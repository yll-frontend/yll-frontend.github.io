```js
all(list) {
  return new Promise((resolve, reject) => {
    const list = Array.from(list)
    const result = []
    list.forEach((promise, index) => {
      Promise.resolve(promise).then(res => {
        result[index] = res
        if (index === list.length) {
          resolve(result)
        }
      }).catch(error => {
        reject(error)
      })
    })
  })
}
```
