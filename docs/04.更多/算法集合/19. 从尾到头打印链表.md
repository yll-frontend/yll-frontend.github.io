/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {number[]}
 */
//两个思路
// 解法一 利用栈的特性
var reversePrintreversePrint = function (head) {
  const res = []
  while (head) {
    res.unshift(head.value)
    head = head.next
  }
  return res
};

//递归
var reversePrint = function(head) {
  //递归
  let res = [];
  const reverse = (head) => {
      if(!head) return;
      reverse(head.next);
      res.push(head.val);
  }
  reverse(head);
  return res;
};
