# 递归基础版

function deepClone(target) {
  if (typeof target !== 'object' || target === null) {
    return target
  }
  let result = {}
  for (let key in target) {
    result[key] = deepClone(target[key])
  }
  return result
}


const bar = deepClone(foo)
bar.girlFriend.name = '小梨'

# 递归升级版(兼容数组以及函数)

function deepClone3(target) {
  if (typeof target !== 'object' || target === null) {
    return target
  }
  let result;
  if (Array.isArray(target)) {
    result = []
  } else if (target instanceof Function) {
    result = function () {
      return target.call(this, ...args)
    }
  } else {
    result = {}
  }
  for (let key in target) {
    result[key] = deepClone3(target[key])
  }
  return result
}

# 递归终极版(兼容所有类型，eg：Date,Regexp等)



function deepClone4(target) {
  if (typeof target !== 'object' || target === null) {
    return target
  }
  let result;
  if (Array.isArray(target)) {
    result = []
  } else if (target instanceof Function) {
    result = function () {
      return target.call(this, ...args)
    }
  } else if (target instanceof Date) {
    result = new Date(target)
  } else if (target instanceof Regexp) {
    //克隆正则，就是克隆其source以及falgs属性
    result = new Regexp(target.source, target.flags)
  }
  else {
    result = {}
  }
  for (let key in target) {
    result[key] = deepClone3(target[key])
  }
  return result
}

  • 循环引用 可以利用map记录下已经拷贝过的属性
const cache = new Map()
function deepClone4(target) {
  if(cache.get(target)){
      return cache.get(target)
  }
  if (typeof target !== 'object' || target === null) {
    return target
  }
  let result;
  if (Array.isArray(target)) {
    result = []
  } else if (target instanceof Function) {
    result = function () {
      return target.call(this, ...args)
    }
  } else if (target instanceof Date) {
    result = new Date(target)
  } else if (target instanceof Regexp) {
    //克隆正则，就是克隆其source以及falgs属性
    result = new Regexp(target.source, target.flags)
  } else {
    result = {}
  }
  //记录在Map中
  cache.set(target, result);
  for (let key in target) {
    //过滤原型上的属性
    if(target.hasOwnProperty(key)) {
      result[key] = deepClone3(target[key])
    }
  }
  return result
}

这种方式会直接导致一个问题，即多次拷贝对象时，会导致缓存越来越多，并且，可能多个对象有相同的属性，会直接出现错误,这个问题可以通过每次拷贝时，新建一个Map对象解决

function deepClone(target,catch = new Map()) {
  if(cache.get(target)){
      return cache.get(target)
  }
  if (typeof target !== 'object' || target === null) {
    return target
  }
  let result;
  if (Array.isArray(target)) {
    result = []
  } else if (target instanceof Function) {
    result = function () {
      return target.call(this, ...args)
    }
  } else if (target instanceof Date) {
    result = new Date(target)
  } else if (target instanceof Regexp) {
    //克隆正则，就是克隆其source以及falgs属性
    result = new Regexp(target.source, target.flags)
  } else {
    result = {}
  }
  //记录在Map中
  cache.set(target, result);
  for (let key in target) {
    //过滤原型上的属性
    if(target.hasOwnProperty(key)) {
      result[key] = deepClone(target[key],catch)
    }
  }
  return result
}
